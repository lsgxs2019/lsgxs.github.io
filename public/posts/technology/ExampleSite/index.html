$ mkdir your-site
$ cd your-site
$ mkdir themes
$ cd themes
$ git clone https://github.com/repository-url.git
$ cp -r themes-name/exampleSite/** ../    --把ExampleSite站点目录下的所有内容复制到你的站点根目录
$ cd ..
$ hugo serve
